// --- Trig Precision Drift Test ---
// This script forces agents into a continuous, tight circular orbit. 
// Any floating-point differences in trig approximations between 
// the CPU and GPU will cause the orbits to visually separate over time.

input speed = 1.0;
input turnAngle = 0.05;

// 1. Turn by a constant, small angle every frame.
// This forces continuous evaluation of sine/cosine under the hood.
turn(inputs.turnAngle);

// 2. Move forward applies the new angle to the velocity vector.
moveForward(inputs.speed);
